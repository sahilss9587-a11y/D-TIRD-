<script>
let mines=[]
function startMines(){
mines=[]
for(let i=0;i<3;i++){mines.push(Math.floor(Math.random()*9))}
let html='<h2>Mines Game</h2><div class="grid">'
for(let i=0;i<9;i++){
html+=`<div class="box" onclick="clickMine(${i},this)">?</div>`
}
html+='</div>'
document.getElementById("gameArea").innerHTML=html
}
function clickMine(i,el){
if(mines.includes(i)){
el.innerHTML="💣"alert("Game Over")
}else{
el.innerHTML="✅"
}
}
function startChicken(){
let result=Math.random()
if(result>0.5){
alert("🐔 Chicken crossed road successfully!")
}else{
alert("🚗 Chicken got hit!")
}
}
function startPenalty(){
let player=Math.floor(Math.random()*3)let keeper=Math.floor(Math.random()*3)
if(player==keeper){
alert("🧤 Goalkeeper Saved")
}else{
alert("⚽ GOAL!")
}
}
